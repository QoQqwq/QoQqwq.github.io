import numpy as np
from cv2 import cv2
sdThresh = 5.8
font = cv2.FONT_HERSHEY_SIMPLEX
face_cascade = cv2.CascadeClassifier('D:/app\code\python\Lib\site-packages\cv2\data\haarcascade_frontalface_alt.xml')
eye_cascade = cv2.CascadeClassifier('D:/app\code\python\Lib\site-packages\cv2\data\haarcascade_eye.xml')
def distMap(frame1, frame2):

    frame1_32 = np.float32(frame1) #读取照片np读取三维数组数据
    frame2_32 = np.float32(frame2)
    #frame3_32 = np.float32(frame3)    
    diff321 = frame1_32 - frame2_32 #数据处理和一些小实验
    #diff322 = frame2_32 - frame3_32
    #diff323 = diff321 + diff322#=1-3 
    norm32 = np.sqrt(diff321[:,:,0]**2 + diff321[:,:,1]**2 + \
        diff321[:,:,2]**2)/np.sqrt(255**2 + 255**2 + 255**2)#滤波
    dist = np.uint8(norm32*255)#数据格式转换
    return dist
cv2.namedWindow('frame')#创建窗口
cv2.namedWindow('dist')
cap = cv2.VideoCapture(0)#参数0表示从笔记本使用摄像头
_, frame1 = cap.read()#这里使提取摄像头的图片
_, frame2 = cap.read()
x1 = 0
facecount = 0
while(True):
    _, frame3 = cap.read()
    rows, cols, _ = np.shape(frame3)
    cv2.imshow('dist', frame3)
    dist = distMap(frame1, frame3)
    frame1 = frame2
    frame2 = frame3
    mod = cv2.GaussianBlur(dist, (9,9), 0)#高斯滤波
    _, thresh = cv2.threshold(mod, 100, 255, 0)
    _, stDev = cv2.meanStdDev(mod)#处理图片
    cv2.imshow('dist', mod)#调出图片
    cv2.putText(frame2, "Standard Deviation - {}".format(round(stDev[0][0],0)), \
                (70, 70), font, 1, (255, 0, 255), 1, cv2.LINE_AA)
    if stDev > sdThresh:#即处理后的数值大于设定的数值
            print(x1)
            faces = face_cascade.detectMultiScale(frame3,1.2,2)
            for (x,y,w,h) in faces:#根据自建库画出框框
                cv2.rectangle(frame3,(x,y),(x+w,x+h),(255,0,0),2)
                face_re = frame3[y:y+h,x:x+w]
                eyes = eye_cascade.detectMultiScale(face_re)
                for (ex,ey,ew,eh) in eyes:
                    cv2.rectangle(face_re,(ex,ey),(ex+ew,ey+eh),(0,255,0),2)
            x1=x1+1
    cv2.imshow('frame', frame2)
    if cv2.waitKey(1) & 0xFF == 27:#这里比较难懂，基本上使退出操作
        break

#import tensorflow, 無論 CPU 或 GPU 版本都是 import tensorflow as tf
import tensorflow as tf
#將MNIST 手寫數字資料讀進來
mnist = tf.keras.datasets.mnist
# mnist 的load_data()會回傳已經先分割好的training data 和 testing data
# 並且將每個 pixel 的值從 Int 轉成 floating point 同時做normalize(這是很常見的preprocessing)
(x_train, y_train), (x_test, y_test) = mnist.load_data()
x_train, x_test = x_train / 255.0, x_test / 255.0
print(len(x_train)) #training data 總共有60000張圖片
print(x_train[0].shape) #每張圖片（拿第一張當樣本）大小為 28x28
# 開始搭建model
# 利用 "Sequential" 把每層 layer 疊起來

# input 大小為 28 x 28

# 最後的 Dense(10) 且 activation 用 softmax
# 代表最後 output 為 10 個 class （0~9）的機率
model = tf.keras.models.Sequential([
tf.keras.layers.Flatten(input_shape=(28, 28)),
tf.keras.layers.Dense(128, activation='relu'),
tf.keras.layers.Dropout(0.2),
tf.keras.layers.Dense(10, activation='softmax')
])

# model每層定義好後需要經過compile
model.compile(optimizer='adam',
            loss='sparse_categorical_crossentropy',
            metrics=['accuracy'])
# 將搭好的 model 去 fit 我們的 training data
# 並evalutate 在 testing data 上
model.fit(x_train, y_train, epochs=5)
model.evaluate(x_test, y_test, verbose=2)
